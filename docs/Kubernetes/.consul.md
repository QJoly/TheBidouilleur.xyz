


```bash
helm repo add hashicorp https://helm.releases.hashicorp.com
helm install --values consul-values.yml consul hashicorp/consul --create-namespace --namespace consul
```

```yml
# values.yml
global:
 enabled: true
 name: consul
 datacenter: dc1
 tls:
   enabled: true
 acls:
   manageSystemACLs: true
server:
 enabled: true
 replicas: 1
ui:
 enabled: true
 service:
   type: NodePort
connectInject:
 enabled: true
```
